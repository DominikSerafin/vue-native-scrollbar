  var app = new Vue({

    el: '#app',

    data: function(){
      return {
        popoverOpen: false,
        daySeconds: 0,
      }
    },

    created: function(){
      this.setDaySeconds();
    },

    mounted: function(){
      this.daySecondsInterval();
    },

    methods: {

      setDaySeconds: function(){
        var dt = new Date();
        var secs = dt.getSeconds() + (60 * dt.getMinutes()) + (60 * 60 * dt.getHours());
        this.daySeconds = secs;
      },

      daySecondsInterval: function(){
        setInterval(function () {
          this.setDaySeconds();
        }.bind(this), 1000);
      },

    },

  });
